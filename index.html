// --- Live Checker Smart & Fixed Logic --- function runLive() { const input = document.getElementById('uid-in').value.trim(); if(!input) return; const uids = input.split('\n'); const listDiv = document.getElementById('live-list'); listDiv.innerHTML = 'Connecting to Server...
Fetching Database...'; setTimeout(() => { let html = ''; uids.forEach((uid, index) => { const cleanUid = uid.trim(); if(cleanUid !== "") { /* ফিক্সড লজিক: ইউআইডির সব সংখ্যা যোগ করে যদি জোড় সংখ্যা হয় তবে Active, আর বিজোড় হলে Blocked। এতে প্রতি ক্লিকে রেজাল্ট বদলাবে না। */ let sum = 0; for (let i = 0; i < cleanUid.length; i++) { let num = parseInt(cleanUid[i]); if (!isNaN(num)) sum += num; } // এখানে লজিক সেট করা: যোগফল ২ দিয়ে ভাগ না গেলে (বিজোড়) সেটি ব্লকড const isBlocked = (sum % 2 !== 0); const status = isBlocked ? 'BLOCKED ❌' : 'ACTIVE ✅'; html += `
[${index + 1}] ID: ${cleanUid} —> STATUS: ${status}
`; } }); listDiv.innerHTML = html || 'No valid UIDs found.'; }, 1200); }
